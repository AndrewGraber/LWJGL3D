#This is the configuration file. If you muck this all up, just delete the file.
#It will be recreated with default values.

#The Width of the screen (Default: 1280)
i:screen_width=1280

#The height of the screen (Default: 720)
i:screen_height=720

#The maximum value for the FPS (Default: 60)
i:fps_cap=60

#The Sensitivity of the mouse (Default: 0.25f)
f:mouse_sensitivity=0.25f

#The Sensitivity of the controller joysticks (Default: 1.0f)
f:controller_sensitivity=1.0f

#The distance one scroll of the mouse wheel (or controller zoom)
#moves the camera (Default: 0.1f)
f:zoom_distance=0.1f

#The amount (in degrees) to change pitch of follow camera by for each
#pixel moved (Default: 0.1f)
f:pitch_change=0.1f

#The amount (in degrees) to change rotation of follow camera around player
#for each pixel moved (Default: 0.3f)
f:rotate_change=0.3f

#The Field of View of the camera (Default: 70f)
f:camera_fov=70f

#The distance of the near clipping plane (Default: 0.1f)
f:near_clip=0.1f

#The render distance of the camera (Default: 1000f)
f:render_distance=1000f

#The Movement speed of the camera (Default: 0.1f)
f:camera_move_speed=0.1f

#The Movement speed of the player (Default: 0.2f)
f:player_move_speed=0.2f

#The ambient light (brightness) in the world. This is the
#amount of light that will show when no light source is in
#range. 0.0f would be complete darkness. 1.0f would be complete
#light (no shadows). (Default: 0.2f)
f:ambient_light=0.2f

#The density of the fog that slowly fades objects away (Default: 0.0035f)
f:fog_density=0.0035f

#The gradient of the fog (Default: 5f)
f:fog_gradient=5f

#The color of the sky - in red, green, and blue in that order.
#(Default: 0.7f, 0.9f, 1.0f)
f:sky_red=0.7f
f:sky_green=0.9f
f:sky_blue=1.0f

#The mipmapping level (just leave if you don't understand) (Default: -0.4f)
f:mipmap_level=-0.4f
